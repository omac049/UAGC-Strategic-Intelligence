document.addEventListener('DOMContentLoaded', function() {
  // Check if this is first visit
  if (!getCookie('first_visit')) {
    // Set first visit cookie (90-day expiration)
    const expDate = new Date();
    expDate.setDate(expDate.getDate() + 90);
    document.cookie = `first_visit=${Date.now()};expires=${expDate.toUTCString()};path=/;SameSite=Strict`;
    
    // Initialize journey stage
    document.cookie = `journey_stage=awareness;expires=${expDate.toUTCString()};path=/;SameSite=Strict`;
    
    // Capture UTM parameters if present
    const utmParams = getUTMParameters();
    if (Object.keys(utmParams).length > 0) {
      document.cookie = `utm_data=${JSON.stringify(utmParams)};expires=${expDate.toUTCString()};path=/;SameSite=Strict`;
    }
    
    // Push to dataLayer for GA4
    window.dataLayer = window.dataLayer || [];
    window.dataLayer.push({
      'event': 'first_visit',
      'journey_stage': 'awareness',
      'utm_data': utmParams
    });
  }
});

    // Dynamic Content Personalization Implementation
class ContentPersonalizer {
  constructor() {
    this.utmParams = this.getUTMParameters();
    this.journeyData = this.getJourneyData();
    this.personalize();
  }
  
  // Get UTM parameters from URL or cookie
  getUTMParameters() {
    // First check URL
    const params = new URLSearchParams(window.location.search);
    const utmParams = {};
    
    ['utm_source', 'utm_medium', 'utm_campaign', 'utm_content', 'utm_term'].forEach(param => {
      if (params.has(param)) {
        utmParams[param] = params.get(param);
      }
    });
    
    // If no UTMs in URL, check cookie
    if (Object.keys(utmParams).length === 0) {
      const utmCookie = document.cookie
        .split('; ')
        .find(row => row.startsWith('uagc_utm_data='));
        
      if (utmCookie) {
        try {
          return JSON.parse(decodeURIComponent(utmCookie.split('=')[1]));
        } catch (e) {
          console.error('Error parsing UTM cookie:', e);
          return {};
        }
      }
    }
    
    return utmParams;
  }
  
  // Get journey data from cookie
  getJourneyData() {
    const journeyCookie = document.cookie
      .split('; ')
      .find(row => row.startsWith('uagc_visitor_journey='));
      
    if (journeyCookie) {
      try {
        return JSON.parse(decodeURIComponent(journeyCookie.split('=')[1]));
      } catch (e) {
        console.error('Error parsing journey cookie:', e);
        return null;
      }
    }
    
    return null;
  }
  
  // Personalize content based on UTM and journey data
  personalize() {
    this.personalizeByUTM();
    this.personalizeByJourneyStage();
    this.personalizeByInterests();
    this.prePopulateForms();
  }
  
  // Personalize based on UTM parameters
  personalizeByUTM() {
    if (!this.utmParams || Object.keys(this.utmParams).length === 0) return;
    
    // Campaign-specific personalization
    if (this.utmParams.utm_campaign) {
      // Military campaign personalization
      if (this.utmParams.utm_campaign.includes('military')) {
        this.showMilitaryContent();
      }
      
      // Program-specific personalization
      if (this.utmParams.utm_campaign.includes('MBA')) {
        this.highlightProgram('MBA');
      } else if (this.utmParams.utm_campaign.includes('healthcare')) {
        this.highlightProgram('Healthcare');
      }
    }
    
    // Source-specific personalization
    if (this.utmParams.utm_source) {
      document.querySelectorAll('[data-utm-source]').forEach(element => {
        if (element.getAttribute('data-utm-source') === this.utmParams.utm_source) {
          element.classList.remove('d-none');
        }
      });
    }
  }
  
  // Personalize based on journey stage
  personalizeByJourneyStage() {
    if (!this.journeyData || !this.journeyData.currentStage) return;
    
    const stage = this.journeyData.currentStage;
    
    // Show stage-specific content
    document.querySelectorAll('[data-journey-stage]').forEach(element => {
      if (element.getAttribute('data-journey-stage') === stage) {
        element.classList.remove('d-none');
      } else if (element.getAttribute('data-journey-stage') !== 'all') {
        element.classList.add('d-none');
      }
    });
    
    // Adjust CTAs based on journey stage
    this.updateCTAsByStage(stage);
  }
  
  // Personalize based on detected interests
  personalizeByInterests() {
    if (!this.journeyData || !this.journeyData.pagesViewed) return;
    
    const pagesViewed = this.journeyData.pagesViewed;
    
    // Check for military interest
    if (pagesViewed.some(page => page.includes('/military'))) {
      this.showMilitaryContent();
    }
    
    // Check for specific program interests
    const programInterests = this.detectProgramInterests(pagesViewed);
    if (programInterests.length > 0) {
      this.highlightProgram(programInterests[0]);
    }
  }
  
  // Detect program interests from pages viewed
  detectProgramInterests(pagesViewed) {
    const interests = [];
    const programPatterns = {
      'MBA': /\/online-degrees\/business\/mba/,
      'Education': /\/online-degrees\/education/,
      'Healthcare': /\/online-degrees\/health/,
      'Psychology': /\/online-degrees\/psychology/
    };
    
    Object.entries(programPatterns).forEach(([program, pattern]) => {
      if (pagesViewed.some(page => pattern.test(page))) {
        interests.push(program);
      }
    });
    
    return interests;
  }
  
  // Show military-specific content
  showMilitaryContent() {
    document.querySelectorAll('[data-interest="military"]').forEach(element => {
      element.classList.remove('d-none');
    });
    
    // Update hero image if it exists
    const heroImage = document.querySelector('.hero-image');
    if (heroImage) {
      heroImage.src = '/images/military-hero.jpg';
      heroImage.alt = 'UAGC Military Student Success';
    }
    
    // Update headline if it exists
    const headline = document.querySelector('.hero-headline');
    if (headline) {
      headline.textContent = 'Military Education Benefits at UAGC';
    }
  }
  
  // Highlight a specific program
  highlightProgram(program) {
    document.querySelectorAll('[data-program]').forEach(element => {
      if (element.getAttribute('data-program') === program) {
        element.classList.remove('d-none');
        
        // Move this program to the top of any program lists
        const programList = element.parentElement;
        if (programList && programList.classList.contains('program-list')) {
          programList.prepend(element);
        }
      }
    });
  }
  
  // Update CTAs based on journey stage
  updateCTAsByStage(stage) {
    const ctaMap = {
      'awareness': 'Learn More',
      'consideration': 'Request Information',
      'engagement': 'Apply Now',
      'application': 'Complete Your Application'
    };
    
    const ctaText = ctaMap[stage] || 'Learn More';
    
    document.querySelectorAll('.dynamic-cta').forEach(cta => {
      cta.textContent = ctaText;
    });
  }
  
  // Pre-populate form fields based on stored data
  prePopulateForms() {
    if (!this.journeyData) return;
    
    // Find RFI and application forms
    document.querySelectorAll('form[data-form-type]').forEach(form => {
      // Email field
      const emailField = form.querySelector('input[type="email"]');
      if (emailField && this.journeyData.email) {
        emailField.value = this.journeyData.email;
      }
      
      // Program interest field
      const programField = form.querySelector('select[name="program_interest"]');
      if (programField && this.journeyData.programInterest) {
        programField.value = this.journeyData.programInterest;
      }
      
      // Military status field
      const militaryField = form.querySelector('select[name="military_status"]');
      if (militaryField && this.journeyData.isMilitary) {
        militaryField.value = 'active_duty';
      }
    });
  }
